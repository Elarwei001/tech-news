interface SpokenOutput {
  spoken: string;  // 必须是 {"spoken":"..."} 格式
}

function extractSpokenOutput(response: string): string {
  // 优先级：直接 JSON > fenced JSON > inline fallback
  
  // 1. 尝试直接解析 JSON
  const direct = tryParseJSON(response);
  if (direct?.spoken) return sanitize(direct.spoken);
  
  // 2. 尝试提取 fenced JSON
  const fenced = extractFencedJSON(response);
  if (fenced?.spoken) return sanitize(fenced.spoken);
  
  // 3. 抑制 reasoning/meta/error 内容

}

async function playAudio(callSid: string, audio: Buffer) {
  const stream = this.activeStreams.get(callSid);
  
  if (stream) {
    // 必须使用流式 TTS
    const result = await stream.sendTTS(audio);
    if (!result.success) {
      // 失败：不回退到 TwiML
      throw new StreamPlaybackError(result.droppedChunks);
    }
  } else {
    // 无活跃流：允许 TwiML <Say> fallback
    await this.twimlSay(callSid, audio);
  }
}

// 流重连正确性
function unregisterStream(callSid: string, streamSid: string) {
  const current = this.activeStreams.get(callSid);
  // 仅当 SID 匹配时才移除，防止新流被误删
  if (current?.sid === streamSid) {
    this.activeStreams.delete(callSid);
  }
}

### 1. ACP Session 卡死恢复 [#51816]

**PR**: [ACP: recover hung bound turns](https://github.com/openclaw/openclaw/pull/51816)  
**作者**: @dutifulbob | **规模**: M (+454/-21)

**问题**：当一个 `runtime.runTurn()` 永不完成时，该 session 上的所有后续 turn 都会阻塞在队列中，导致 Discord ACP binding（如 `#codex-1`）看起来"死了"。

**修复**：
```typescript
// manager.core.ts — 管理端 watchdog
async function runTurn(sessionKey: string, message: string) {
  const timeout = this.getSessionTimeout(sessionKey);
  
  const result = await Promise.race([
    this.runtime.runTurn(sessionKey, message),
    this.createWatchdog(timeout)
  ]);
  
  if (result === WATCHDOG_TIMEOUT) {
    // 超时处理
    log.warn("ACP turn timed out", { sessionKey, timeout });
    
    // 1. 中止/取消/关闭超时的 turn
    await this.abortTurn(sessionKey);
    
    // 2. 丢弃缓存的 runtime handle
    this.cachedRuntimes.delete(sessionKey);
    
    // 3. 返回错误让队列中的 turn 继续
    throw new AcpTurnTimeoutError(timeout);
  }
  
  return result;
}
```

**影响**：单个卡死的 turn 不再能阻塞整个 session。超时后清理并允许后续 turn 继续。
